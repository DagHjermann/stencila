# A R snippet (from the dplyr vignette)

by_tailnum <- group_by(flights, tailnum)
delay <- summarise(by_tailnum,
  count = n(),
  dist = mean(distance, na.rm = TRUE),
  delay = mean(arr_delay, na.rm = TRUE))
delay <- filter(delay, count > 20, dist < 2000)

# Interestingly, the average delay is only slightly related to the
# average distance flown by a plane.
ggplot(delay, aes(dist, delay)) +
  geom_point(aes(size = count), alpha = 1/2) +
  geom_smooth() +
  scale_size_area()

// Javascript; totally meta!

'use strict';

var CodeEditorComponent = require('../../ui/CodeEditorComponent');


function CodeblockComponent() {
  CodeblockComponent.super.apply(this, arguments);

  this.editor = null;
}

CodeblockComponent.Prototype = function() {

  var _super = CodeblockComponent.super.prototype;

  this.render = function($$) {
    var el = _super.render.call(this, $$);
    el.addClass('sc-codeblock');
    return el;
  };

};

CodeEditorComponent.extend(CodeblockComponent);

CodeblockComponent.fullWidth = true;

module.exports = CodeblockComponent;

import unittest
import sys

import stencila

class ExceptionTests(unittest.TestCase):

    def test_translation(self):
        try:
            stencila.extension.exception_test()
        except Exception, e:
            self.assertEqual(str(e), "Testing, testing, 1, 2, 3.")
        else:
            raise


class ComponentTests(unittest.TestCase):

    def setUp(self):
        self.component = stencila.Component()

    def tearDown(self):
        # As in C++ tests the destoy method fails due to
        # "Acesss is denied" to the .git directory of the
        # component. So currently skipped in Windows
        if sys.platform != 'win32':
            self.component.destroy()
