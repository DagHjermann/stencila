  count = n(),
  dist = mean(distance, na.rm = TRUE),
  delay = mean(arr_delay, na.rm = TRUE))

# Interestingly, the average delay is only slightly related to the

  geom_point(aes(size = count), alpha = 1/2) +
  geom_smooth() +
